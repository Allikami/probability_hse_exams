
\begin{question}
Пусть \(X = (X_1, \ldots , X_n)\) — случайная выборка из биномиального
распределения \(Bi(5, p)\). Известно, что
\(\P(X = x) =C_n^x p^x(1-p)^{n-x}\). Информация Фишера \(I_n(p)\) равна:
\begin{answerlist}
  \item \(\frac{5p(1-p)}{n}\)
  \item \(\frac{n}{5p(1-p)}\)
  \item \(\frac{n}{p(1-p)}\)
  \item \(\frac{5n}{p(1-p)}\)
  \item \(\frac{p(1-p)}{5n}\)
\end{answerlist}
\end{question}

\begin{solution}
\begin{answerlist}
  \item Неверно
  \item Неверно
  \item Неверно
  \item Отлично
  \item Неверно
\end{answerlist}
\end{solution}
