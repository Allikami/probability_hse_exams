  \item \(-\E \left( \frac{\partial \ell (\theta)}{\partial \theta}\right)\)

  \item \(-\E \left( \left( \frac{\partial \ell (\theta)}{\partial \theta} \right)^2 \right)\)
\end{answerlist}
\end{question}
