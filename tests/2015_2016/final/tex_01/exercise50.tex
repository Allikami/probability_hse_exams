
\begin{question}

называется состоятельной, если
\begin{answerlist}
  \item \(\Var(\hat\theta_n) \to 0\)
  \item \(\Var(\hat\theta_n) \geq \Var(\hat\theta_n + 1)\)
  \item \(\P(|\hat\theta_n - \theta | > t) \to 0\) для всех \(t > 0\)
  \item \(\E(\hat\theta_n) \to \theta\)
  \item \(\E(\hat\theta_n) = \theta\)
\end{answerlist}
\end{question}

\begin{solution}
\begin{answerlist}
  \item Неверно
  \item Неверно
  \item Отлично
  \item Неверно
  \item Неверно
\end{answerlist}
\end{solution}
