
\begin{question}
Пусть \(X = (X_1, \ldots , X_n)\) — случайная выборка из распределения
с плотностью \[
f(x; \theta) =
\begin{cases}
\frac{1}{\theta}\exp(-\frac{x}{\theta}) \text{ при } x \geq 0,  \\
0 \text{ при } x < 0.
\end{cases}
\] При каком значении константы \(c\) оценка
\(\hat{\theta} = c \bar{X}\) является несмещённой?
\begin{answerlist}
  \item \(\frac{n}{n+1}\)
  \item \(\frac{1}{n}\)
  \item \(\frac{n + 1}{n}\)
  \item \(1\)
  \item \(n\)
\end{answerlist}
\end{question}

\begin{solution}
\begin{answerlist}
  \item Неверно
  \item Неверно
  \item Неверно
  \item Отлично
  \item Неверно
\end{answerlist}
\end{solution}
