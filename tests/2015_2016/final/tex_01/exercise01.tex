дисперсиях, тестовая статистика имеет распределение
\begin{answerlist}
  \item \(t_{m+n-1}\)
  \item \(t_{m+n-2}\)
  \item \(\cN(0;1)\)
  \item \(F_{m-1,n-1}\)
  \item \(F_{m}\)
\end{answerlist}
\end{question}

\begin{solution}
\begin{answerlist}
  \item Неверно
  \item Неверно
  \item Отлично
  \item Неверно
  \item Неверно
\end{answerlist}
\end{solution}
