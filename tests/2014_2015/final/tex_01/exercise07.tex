верной \(H_0\)

  \item \(LR=\ln 34\), \(\chi^2_{n-2}\)
  \item \(LR=34\), \(\chi^2_2\)
  \item \(LR=68\), \(\chi^2_2\)
  \item \(LR=34\), \(\chi^2_{n-1}\)
\end{answerlist}
\end{question}
