  \item оценкой максимального правдоподобия
  \item эффективной оценкой \(p^2\)
  \item состоятельной оценкой \(p^2\)
  \item несмещенной оценкой \(p^2\)
  \item асимптотически нормальной оценкой \(p^2\)
\end{answerlist}
\end{question}

\begin{solution}
\begin{answerlist}
  \item Неверно
  \item Неверно
  \item Неверно
  \item Отлично
  \item Неверно
\end{answerlist}
\end{solution}
